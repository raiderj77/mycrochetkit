┌─────────────────────────────────────────────────────────────────┐
│ 🚀 FAST DEPLOY (30 seconds)                                     │
└─────────────────────────────────────────────────────────────────┘
npm run build
firebase deploy --only hosting
git add . && git commit -m "message" && git push

┌─────────────────────────────────────────────────────────────────┐
│ 🏭 PRODUCTION DEPLOY (2-3 minutes, auto)                        │
└─────────────────────────────────────────────────────────────────┘
git add .
git commit -m "Release: description"
git push origin main
# Cloud Build does the rest automatically

┌─────────────────────────────────────────────────────────────────┐
│ 🔧 DAILY WORKFLOW                                               │
└─────────────────────────────────────────────────────────────────┘
# Start of day
cd ~/mycrochetkit && git pull origin main

# During work (repeat as needed)
npm run build && firebase deploy --only hosting

# End of day
git add . && git commit -m "Summary" && git push

┌─────────────────────────────────────────────────────────────────┐
│ 🆘 EMERGENCY COMMANDS                                           │
└─────────────────────────────────────────────────────────────────┘
git status                    # What changed?
firebase projects:list        # Which project am I on?
firebase use my-crochetkit    # Switch to production
git reset --hard HEAD         # Discard all changes (DANGER!)
git reset --soft HEAD~1       # Undo last commit (keep changes)

└─────────────────────────────────────────────────────────────────┘
